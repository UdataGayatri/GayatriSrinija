/** Program to find sum of n integers Eg: sum of 10 integers=1+2+3+4+…….10=55**/
#include<stdio.h>
int main()
{
	int n,sum,i;
	i = 1;
	sum = 0;
	printf("enter any number:\n");
	scanf("%d",&n);
    while(i<=n)
	{
		sum = sum + i;
		i++;
	}
	printf("the sum of n integers is : %d",sum); 	
	return 0;
}
